cmake_minimum_required(VERSION 3.5)
project(Kingdom-Fall)

set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_FLAGS_DEBUG "-g -fsanitize=address -O0 -Wall")
set(CMAKE_CXX_FLAGS_RELEASE "-O3")
set(CMAKE_CXX_FLAGS ${CMAKE_CXX_FLAGS_RELEASE})
if (DEBUG STREQUAL "ON")
    set(CMAKE_CXX_FLAGS ${CMAKE_CXX_FLAGS_DEBUG})
endif ()
add_library(json-hpp INTERFACE)
target_include_directories(json-hpp
        INTERFACE
        third_party/)
SET(SDL2PP_WITH_IMAGE ON)
SET(SDL2PP_WITH_TTF ON)
ADD_SUBDIRECTORY(third_party/libSDL2pp)

add_subdirectory(third_party/json-schema-validator json-schema-validator)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR})
set(SOURCE_FILES src/CObject.cpp src/CObject.hpp src/CPosition.cpp src/CPosition.hpp src/CUnit.cpp src/CUnit.hpp src/CTerrain.cpp src/CTerrain.hpp src/CUnitFactoryBuilder.cpp src/CUnitFactoryBuilder.hpp src/defines.hpp src/json.hpp src/json-validator.hpp src/CObjectFactory.hpp src/CVillage.cpp src/CVillage.hpp src/CMapCell.cpp src/CMapCell.hpp src/CMap.cpp src/CMap.hpp src/CJsonSerializerAdapter.cpp src/CJsonSerializerAdapter.hpp src/CurrentSerializer.hpp src/CJsonValidatorAdapter.cpp src/CJsonValidatorAdapter.hpp src/CObjectFactoryValidateDecorator.hpp src/IObjectFactory.hpp src/CGlobalGame.cpp src/CGlobalGame.hpp src/CObjectPositionView.cpp src/CObjectPositionView.hpp src/IObjectObserver.hpp src/IObjectObserver.cpp src/CObjectController.cpp src/CObjectController.hpp src/IObjectController.cpp src/IObjectController.hpp src/CControllerFactory.cpp src/CControllerFactory.hpp src/CObjectFactory.cpp src/CObjectFactoryValidateDecorator.cpp src/CObjectProxyView.cpp src/CObjectProxyView.hpp src/IControllerFactory.hpp src/CSelectCommand.cpp src/CSelectCommand.hpp src/CObjectSelectView.cpp src/CObjectSelectView.hpp src/CCommand.cpp src/CCommand.hpp src/IObjectView.cpp src/IObjectView.hpp)
set(GAME_SOURCES ${SOURCE_FILES} src/main.cpp)
add_executable(MyGame ${GAME_SOURCES})
target_include_directories(MyGame PUBLIC third_party)
target_include_directories(MyGame PUBLIC ${SDL2PP_INCLUDE_DIRS})
TARGET_LINK_LIBRARIES(MyGame ${SDL2PP_LIBRARIES})
target_link_libraries(MyGame gtest)
if (UNIX)
    target_link_libraries(MyGame pthread)
endif ()
target_link_libraries(MyGame json-schema-validator)
if (WITH_TEST STREQUAL "ON")
    set(TEST_SOURCES ${SOURCE_FILES} test/test.hpp test/test.cpp )
    add_executable(MyTest ${TEST_SOURCES})
    target_include_directories(MyTest PUBLIC third_party)
    target_include_directories(MyTest PUBLIC ${SDL2PP_INCLUDE_DIRS})
    TARGET_LINK_LIBRARIES(MyTest ${SDL2PP_LIBRARIES})
    target_link_libraries(MyTest gtest)
    if (UNIX)
        target_link_libraries(MyTest pthread)
    endif ()
    target_link_libraries(MyTest json-schema-validator)
endif ()
